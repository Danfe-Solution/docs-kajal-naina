
## Mermaid ERD Diagram

```mermaid
erDiagram
    Vendors {
        int vendor_id PK
        string name
        int ledger_id FK
        bool is_active
        datetime created_at
    }

    CustomerTypes {
        int customer_type_id PK
        string name
        string description
    }

    Customers {
        int customer_id PK
        string name
        int customer_type_id FK
        int ledger_id FK
        bool is_active
        datetime created_at
    }

    Suppliers {
        int supplier_id PK
        string name
        int ledger_id FK
        bool is_active
        datetime created_at
    }

    Employees {
        int employee_id PK
        string name
        int ledger_id FK
        bool is_active
        datetime created_at
    }

    Ledgers {
        int ledger_id PK
        string name
        string type
        string code
        bool is_active
    }

    CustomerTypes ||--o{ Customers : "categorizes"
    Ledgers ||--o{ Customers : "linked"
    Ledgers ||--o{ Suppliers : "linked"
    Ledgers ||--o{ Employees : "linked"
    Ledgers ||--o{ Vendors : "linked"
